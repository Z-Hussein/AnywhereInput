import asyncio
import websockets
import json
import pyautogui
import hashlib
import hmac
import secrets
import time
from datetime import datetime
import socket
import netifaces
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class SecureMouseServer:
    def __init__(self):
        self.clients = {}
        self.trusted_tokens = {}
        self.screen_width, self.screen_height = pyautogui.size()
        pyautogui.FAILSAFE = False
        
        # Generate server token
        self.server_secret = secrets.token_hex(32)
        self.server_id = hashlib.sha256(self.server_secret.encode()).hexdigest()[:16]
        
        logger.info(f"Server ID: {self.server_id}")
        logger.info(f"Screen: {self.screen_width}x{self.screen_height}")
        
    def generate_client_token(self, client_name):
        """Generate unique token for each client"""
        token = secrets.token_hex(32)
        self.trusted_tokens[token] = {
            'name': client_name,
            'created': datetime.now().isoformat(),
            'last_seen': None,
            'permissions': ['move', 'click', 'scroll', 'keyboard']
        }
        return token
    
    def verify_token(self, token):
        """Verify client token"""
        if token in self.trusted_tokens:
            self.trusted_tokens[token]['last_seen'] = datetime.now().isoformat()
            return True
        return False
    
    async def handle_connection(self, websocket, path):
        """Handle WebSocket connection"""
        client_ip = websocket.remote_address[0]
        logger.info(f"Connection from {client_ip}")
        
        try:
            # Handshake: Client must send token first
            handshake = await asyncio.wait_for(websocket.recv(), timeout=10)
            data = json.loads(handshake)
            
            token = data.get('token')
            client_name = data.get('name', 'Unknown')
            
            if not self.verify_token(token):
                logger.warning(f"Invalid token from {client_ip}")
                await websocket.send(json.dumps({
                    'type': 'error',
                    'message': 'Invalid token'
                }))
                await websocket.close()
                return
            
            logger.info(f"Authenticated: {client_name} from {client_ip}")
            
            # Send connection confirmation with encryption info
            await websocket.send(json.dumps({
                'type': 'handshake_ack',
                'server_id': self.server_id,
                'screen_width': self.screen_width,
                'screen_height': self.screen_height,
                'timestamp': time.time(),
                'permissions': self.trusted_tokens[token]['permissions']
            }))
            
            # Handle commands
            async for message in websocket:
                try:
                    cmd = json.loads(message)
                    await self.execute_command(cmd, websocket)
                except json.JSONDecodeError:
                    await websocket.send(json.dumps({
                        'type': 'error',
                        'message': 'Invalid JSON'
                    }))
                    
        except asyncio.TimeoutError:
            logger.warning(f"Handshake timeout from {client_ip}")
        except websockets.exceptions.ConnectionClosed:
            logger.info(f"Connection closed: {client_ip}")
        except Exception as e:
            logger.error(f"Error: {e}")
    
    async def execute_command(self, cmd, websocket):
        """Execute mouse/keyboard commands"""
        cmd_type = cmd.get('type')
        cmd_id = cmd.get('id', 0)
        
        try:
            if cmd_type == 'move':
                dx = cmd.get('dx', 0)
                dy = cmd.get('dy', 0)
                # Scale movement based on screen
                if cmd.get('mode') == 'absolute':
                    x = int(dx * self.screen_width)
                    y = int(dy * self.screen_height)
                    pyautogui.moveTo(x, y)
                else:
                    pyautogui.moveRel(dx, dy)
                    
            elif cmd_type == 'click':
                button = cmd.get('button', 'left')
                clicks = cmd.get('clicks', 1)
                pyautogui.click(button=button, clicks=clicks)
                
            elif cmd_type == 'double_click':
                pyautogui.doubleClick()
                
            elif cmd_type == 'right_click':
                pyautogui.rightClick()
                
            elif cmd_type == 'scroll':
                amount = cmd.get('amount', 0)
                pyautogui.scroll(amount)
                
            elif cmd_type == 'key':
                key = cmd.get('key', '')
                pyautogui.press(key)
                
            elif cmd_type == 'key_combination':
                keys = cmd.get('keys', [])

Rate limiting:

python
from collections import defaultdict
command_rate = defaultdict(list)

def check_rate_limit(client_id):
    now = time.time()
    commands = command_rate[client_id]
    commands = [t for t in commands if now - t < 1]  # Last second
    if len(commands) >= 30:  # Max 30 commands/sec
        return False
    commands.append(now)
    command_rate[client_id] = commands
    return True
